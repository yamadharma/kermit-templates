\newfontfamily\cyrillicfont[Script=Russian,Ligatures=TeX]{DejaVu Serif Condensed}
\newfontfamily\cyrillicfontsf[Script=Russian,Ligatures=TeX,Scale=MatchLowercase]{DejaVu Sans Condensed}
\newfontfamily\cyrillicfonttt[Script=Russian,Scale=MatchUppercase]{DejaVu Sans Mono}

\setmainfont[Ligatures=TeX]{DejaVu Serif Condensed}
\setsansfont[Ligatures=TeX,Scale=MatchLowercase]{DejaVu Sans Condensed}
\setmonofont[Scale=MatchLowercase,Scale=MatchUppercase]{DejaVu Sans Mono}
